# Dataframe to store values ​​greater than the average:
df_numbers_maior_mean = pd.DataFrame(columns=['numeros_maiores_média'])

# Applying a loop to traverse the list of 'numbers' elements:
for number in numbers:
    if number > mean_numbers:
        df_numbers_maior_mean = df_numbers_maior_mean.append({'numeros_maiores_média': number}, ignore_index=True)
        
# Presentation of results
print(f"Dataframe com os números da lista 'numbers' maiores que a méda {mean_numbers}")
display(df_numbers_maior_mean)
